LOAD EMPLOYEE
LOAD WORKS_ON
LOAD PROJECT
LOAD DEPARTMENT

a1 <- SELECT Salary >= 30000 FROM EMPLOYEE
Q1 <- PROJECT Ssn, Salary FROM a1


DUP <- SELECT Ssn == Ssn FROM EMPLOYEE
RENAME Ssn TO Ssn1 FROM DUP
RENAME Bdate TO Bdate1 FROM DUP
RENAME Sex TO Sex1 FROM DUP
RENAME Salary TO Salary1 FROM DUP
RENAME Super_ssn TO Super_ssn1 FROM DUP
RENAME Dno TO Dno1 FROM DUP

ANS3 <- CROSS EMPLOYEE DUP
ANS2 <- SELECT Super_ssn == Ssn1 FROM ANS3
ANS1 <- SELECT Bdate == Bdate1 FROM ANS2
Q2 <- PROJECT Ssn, Super_ssn, Bdate FROM ANS1



a3 <- CROSS EMPLOYEE WORKS_ON
b3 <- SELECT Ssn == Essn FROM a3
c3 <- PROJECT Ssn,Pno,Sex FROM b3
d3 <- SELECT Ssn == Ssn FROM c3
RENAME Ssn TO Ssn1 FROM d3
RENAME Pno TO Pno1 FROM d3
RENAME Sex TO Sex1 FROM d3

e3 <- CROSS d3 c3
f3 <- SELECT Pno == Pno1 FROM e3
g3 <- SELECT Sex != Sex1 FROM f3
Q3 <- PROJECT Pno FROM g3



a4 <- CROSS EMPLOYEE WORKS_ON
b4 <- SELECT Ssn == Essn FROM a4
c4 <- PROJECT Ssn,Pno,Super_ssn FROM b4
d4 <- SELECT Ssn == Ssn FROM c4
RENAME Ssn TO Ssn1 FROM d4
RENAME Pno TO Pno1 FROM d4
RENAME Super_ssn TO Super_ssn1 FROM d4

e4 <- CROSS d4 c4
f4 <- SELECT Pno == Pno1 FROM e4
g4 <- SELECT Super_ssn == Super_ssn1 FROM f4
Q4 <- PROJECT Ssn,Super_ssn,Pno FROM g4



a5 <- SELECT Dnum >= 1 FROM PROJECT
b5 <- SELECT Dnum <= 5 FROM a5
c5 <- SELECT Dnum != 2 FROM b5
d5 <- SELECT Dnum != 3 FROM c5
Q5 <- SELECT Dnum != 4 FROM d5
